## 二. bind

```js
Function.prototype.ztbind = function(thisArg,...otherProps){
    thisArg = (thisArg === null || thisArg === undefined ) ? window : Object(thisArg)
    Object.defineProperty(thisArg, "fn" , {
        enumerable: false,
        configurable: true,
        writable: false,
        value: this
    })
    // 利用隐式绑定this到thisArg上
    return (...props) => {
        thisArg.fn(...otherProps, ...props)
    }
}
```
